task.spawn(function()
    while true do
        if not getgenv().StopFarm and lp.Character and lp.Character:FindFirstChild("Humanoid") then
            local toolInHand = lp.Character:FindFirstChildOfClass("Tool")
            if not (toolInHand and (toolInHand.Name:find("Fruit") or toolInHand.ToolTip == "Blox Fruit")) then
                for _, v in pairs(lp.Backpack:GetChildren()) do
                    if v:IsA("Tool") and (v.Name:find("Fruit") or v.ToolTip == "Blox Fruit") then
                        lp.Character.Humanoid:EquipTool(v) break
                    end
                end
            end
        end
        task.wait(0)
    end
end)

local function IsPlayerInUI(targetPlayer)
    if not targetPlayer or getgenv().BlackListPlayer[targetPlayer.Name] then return false end
    
    if lp.Team and lp.Team.Name == "Marines" then
        if targetPlayer.Team and targetPlayer.Team.Name == "Marines" then
            return false 
        end
    end

    local gui = game:GetService("CoreGui"):FindFirstChild("FynixHub_System")
    local list = gui and gui:FindFirstChild("ScrollingFrame", true)
    if list then
        for _, v in pairs(list:GetChildren()) do
            if v:IsA("Frame") and v:FindFirstChildOfClass("TextLabel") and v:FindFirstChildOfClass("TextLabel").Text:find(targetPlayer.Name) then
                return true 
            end
        end
    end
    return false
end

local function GetTwoTargetsFromUI()
    local gui = game:GetService("CoreGui"):FindFirstChild("FynixHub_System")
    local list = gui and gui:FindFirstChild("ScrollingFrame", true)
    local validPlayers = {}
    if list then
        for _, v in pairs(list:GetChildren()) do
            if v:IsA("Frame") and v.Name ~= "UIListLayout" then
                local txt = v:FindFirstChildOfClass("TextLabel")
                if txt then
                    local name = txt.Text:split(" [")[1]
                    local p = game:GetService("Players"):FindFirstChild(name)
                    if p and not getgenv().BlackListPlayer[p.Name] and p.Character and p.Character:FindFirstChild("Humanoid") and p.Character.Humanoid.Health > 0 then
                        if lp.Team and lp.Team.Name == "Marines" and p.Team and p.Team.Name == "Marines" then
                        else
                            table.insert(validPlayers, p)
                        end
                    end
                end
            end
        end
    end
    local selected = {}
    if #validPlayers > 0 then
        targetStartTime = tick()
        table.insert(selected, validPlayers[math.random(1, #validPlayers)])
        if #validPlayers > 1 then
            local p2 = validPlayers[math.random(1, #validPlayers)]
            while p2 == selected[1] do p2 = validPlayers[math.random(1, #validPlayers)] end
            table.insert(selected, p2)
        end
    end
    return selected
end

        end)
    end
end)

        local char = lp.Character
        local root = char and char:FindFirstChild("HumanoidRootPart")

            local needNew = #lastTargets == 0

                    needNew = true
                end
            end
            if needNew then lastTargets = GetTwoTargetsFromUI() end

            if #lastTargets > 0 then
                toggle = not toggle
                local currentT = (toggle and lastTargets[1]) or (lastTargets[2] or lastTargets[1])
                local tRoot = currentT.Character.HumanoidRootPart
                
                if hum.Health == 0 then
                    root.CFrame = tRoot.CFrame
                    root.Velocity = Vector3.new(0,0,0)
                end

                if hum.Health > (hum.MaxHealth * 0.1) then
                    root.CFrame = CFrame.new(root.Position.X, root.Position.Y + 2500, root.Position.Z)
                    root.Velocity = Vector3.new(0,0,0)
                end

                for _, v in pairs(char:GetDescendants()) do

                end
            end
        end
    end)

-- BOUNTY TRACKING
local lp = game:GetService("Players").LocalPlayer
local oldBounty = 0
local bountyValue = nil
local start = tick()

if lp:WaitForChild("leaderstats", 5) then
    bountyValue = lp.leaderstats:FindFirstChild("Bounty/Honor") or lp.leaderstats:FindFirstChild("Bounty")
    if bountyValue then
        oldBounty = bountyValue.Value
        bountyValue.Changed:Connect(function(newVal)
            if newVal > oldBounty then getgenv().LastGain = newVal - oldBounty end
            oldBounty = newVal
        end)
    end
end

-- UI UPDATE LOOP
task.spawn(function()
    while true do
        task.wait(0.3)
        local t = math.floor(tick() - start)

        local targets = _G.lastTargets or lastTargets
        
        if targets and #targets > 0 then
            local targetStrings = {}
            for _, p in pairs(targets) do
                local hp = "N/A"
                if p.Character and p.Character:FindFirstChild("Humanoid") then
                    hp = math.floor(p.Character.Humanoid.Health)
                end
                table.insert(targetStrings, p.Name .. " [" .. hp .. "]")
            end
            
            local finalString = table.concat(targetStrings, " & ")
            targetLabel.Text = "Target: " .. finalString
        else
            targetLabel.Text = "Target: Searching..."
        end

        uptime.Text = "Uptime: "..t.."s"
        if bountyValue then
            current.Text = "Current: "..tostring(bountyValue.Value)
        else
            current.Text = "Current: N/A"
        end
        gain.Text = "Gain: +"..tostring(getgenv().LastGain or 0)
        
        -- Update Status
        if getgenv().StopFarm then
            status.Text = "Status: Stopped"
            status.BackgroundColor3 = Color3.fromRGB(200, 0, 0)
        else
            status.Text = "Status: Farming"
            status.BackgroundColor3 = Color3.fromRGB(0, 255, 180)
        end
    end
end)
